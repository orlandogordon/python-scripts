// Jenkinsfile
//
// Discovers all artifact directories under artifacts/, reads each config.yml,
// and runs `make plan` + `make apply` for each one in parallel.
//
// Each parallel branch:
//   1. Selects the appropriate deployer env folder (development/staging/release)
//   2. Hoists backend.tf.tmp -> deployer/backend.tf (with substitutions)
//   3. Merges env-folder terraform.tfvars.json with artifact's config.yml
//   4. Writes merged result to deployer/terraform.tfvars.json
//   5. Runs terraform via the makefile

pipeline {
    agent none

    options {
        disableConcurrentBuilds()
        timeout(time: 2, unit: 'HOURS')
        buildDiscarder(logRotator(numToKeepStr: '50'))
        timestamps()
        ansiColor('xterm')
    }

    environment {
        // Backend token. Hostname dots become underscores.
        TF_TOKEN_abc_company_name_net = credentials('tf-backend-token')
        TF_IN_AUTOMATION = 'true'
        TF_INPUT = 'false'
    }

    stages {

        // ---------------------------------------------------------------
        // Discover: find all artifacts/*/config.yml, validate them,
        // build the deployments list for parallel execution.
        // ---------------------------------------------------------------
        stage('Discover') {
            agent { label 'terraform' }
            steps {
                checkout scm
                sh 'git submodule update --init --recursive'

                script {
                    def configFiles = sh(
                        returnStdout: true,
                        script: 'find artifacts -mindepth 2 -maxdepth 2 -name config.yml | sort'
                    ).trim()

                    if (!configFiles) {
                        error 'No artifacts/*/config.yml files found. Nothing to deploy.'
                    }

                    def deployments = []
                    for (cfgPath in configFiles.split('\n')) {
                        def cfg = readYaml file: cfgPath
                        def stagingDir = cfgPath
                            .replaceFirst('artifacts/', '')
                            .replaceFirst('/config.yml', '')

                        // Required fields that Jenkins itself needs to route the job
                        def required = ['workspace', 'tenant_short', 'target_short', 'env_folder']
                        for (field in required) {
                            if (!cfg[field]) {
                                error "${cfgPath} is missing required field: ${field}"
                            }
                        }

                        // Validate env_folder points to a real directory in the deployer
                        def envFolderPath = "deployer/${cfg.env_folder}"
                        def envFolderExists = sh(
                            returnStatus: true,
                            script: "test -d ${envFolderPath} && test -f ${envFolderPath}/backend.tf.tmp && test -f ${envFolderPath}/terraform.tfvars.json"
                        )
                        if (envFolderExists != 0) {
                            error "${cfgPath} references env_folder '${cfg.env_folder}' but " +
                                  "${envFolderPath}/{backend.tf.tmp,terraform.tfvars.json} not found"
                        }

                        deployments << [
                            staging:      stagingDir,
                            workspace:    cfg.workspace,
                            tenant_short: cfg.tenant_short,
                            target_short: cfg.target_short,
                            env_folder:   cfg.env_folder,
                            config_path:  cfgPath
                        ]
                    }

                    // Guard against duplicate workspaces (would race against same state)
                    def workspaceNames = deployments.collect { it.workspace }
                    def duplicates = workspaceNames
                        .findAll { w -> workspaceNames.count(w) > 1 }
                        .unique()
                    if (duplicates) {
                        error "Duplicate workspace names across staging dirs: ${duplicates}. " +
                              "Each artifact directory must declare a unique workspace."
                    }

                    // Optional: enforce workspace naming convention
                    def workspacePattern = ~/^azure-policy--(qa|uat|prod)--[a-z_0-9]+--[a-z_0-9]+$/
                    for (dep in deployments) {
                        if (!(dep.workspace ==~ workspacePattern)) {
                            error "Workspace '${dep.workspace}' in ${dep.config_path} does not match " +
                                  "naming convention ${workspacePattern}"
                        }
                    }

                    echo "Discovered ${deployments.size()} deployment(s):"
                    for (dep in deployments) {
                        echo "  - ${dep.staging} -> ${dep.workspace} " +
                             "(tenant: ${dep.tenant_short}, env_folder: ${dep.env_folder})"
                    }

                    env.DEPLOYMENT_COUNT = deployments.size().toString()

                    // Stash source for parallel branches
                    stash name: 'source', useDefaultExcludes: false

                    // Persist deployments list across stages
                    writeJSON file: 'deployments.json', json: deployments, pretty: 2
                    stash name: 'deployments-meta', includes: 'deployments.json'
                }
            }
        }

        // ---------------------------------------------------------------
        // Plan: terraform plan for every deployment in parallel.
        // ---------------------------------------------------------------
        stage('Plan') {
            steps {
                script {
                    def deployments = readDeploymentsList()

                    def planJobs = [:]
                    for (deployment in deployments) {
                        def dep = deployment  // capture for closure
                        planJobs[dep.staging] = {
                            node('terraform') {
                                ws("workspace-${dep.staging}-${env.BUILD_NUMBER}") {
                                    unstash 'source'
                                    prepareDeployerWorkdir(dep)
                                    runTerraform(dep, 'plan')
                                    stash(
                                        name: "plan-${dep.staging}",
                                        includes: 'deployer/terraform.plan',
                                        allowEmpty: false
                                    )
                                }
                            }
                        }
                    }
                    parallel planJobs
                }
            }
        }

        // ---------------------------------------------------------------
        // Approve: manual gate before applying. One approval covers all targets.
        // ---------------------------------------------------------------
        stage('Approve') {
            when {
                anyOf {
                    branch 'main'
                    branch 'release/*'
                }
            }
            steps {
                script {
                    def deployments = readDeploymentsList()
                    def msg = "Apply ${deployments.size()} deployment(s)?\n" +
                              deployments.collect { "  - ${it.staging} -> ${it.workspace}" }.join('\n')
                    timeout(time: 30, unit: 'MINUTES') {
                        input message: msg, ok: 'Apply'
                    }
                }
            }
        }

        // ---------------------------------------------------------------
        // Apply: terraform apply for every deployment in parallel,
        // using the plan file from the Plan stage.
        // ---------------------------------------------------------------
        stage('Apply') {
            when {
                anyOf {
                    branch 'main'
                    branch 'release/*'
                }
            }
            steps {
                script {
                    def deployments = readDeploymentsList()

                    def applyJobs = [:]
                    for (deployment in deployments) {
                        def dep = deployment
                        applyJobs[dep.staging] = {
                            node('terraform') {
                                ws("workspace-${dep.staging}-${env.BUILD_NUMBER}") {
                                    unstash 'source'
                                    // Re-prepare the workdir so backend.tf and tfvars are in place
                                    prepareDeployerWorkdir(dep)
                                    unstash "plan-${dep.staging}"
                                    runTerraform(dep, 'apply')
                                }
                            }
                        }
                    }
                    parallel applyJobs
                }
            }
        }
    }

    post {
        always {
            node('terraform') {
                deleteDir()
            }
        }
        failure {
            echo "Build failed. Check parallel branch logs for which deployment(s) failed."
            // Add Slack/email/Teams notification here if desired
        }
    }
}

// =================================================================
// Helpers
// =================================================================

// Read the deployments list stashed during Discover.
def readDeploymentsList() {
    unstash 'deployments-meta'
    return readJSON(file: 'deployments.json')
}

// Hoist the appropriate env folder's backend + tfvars into the deployer
// working directory, merging the per-deployment config.yml on top.
def prepareDeployerWorkdir(Map dep) {
    echo "Preparing deployer workdir for ${dep.staging} using env_folder '${dep.env_folder}'"

    // 1. Render backend.tf from template.
    //    Substitute placeholders in the .tmp file as needed for your setup.
    def backendTemplate = readFile file: "deployer/${dep.env_folder}/backend.tf.tmp"
    def backendRendered = backendTemplate
        .replace('##WORKSPACE##', dep.workspace)
        // Add other substitutions if your template needs them, e.g.:
        // .replace('##HOSTNAME##', 'abc.company_name.net')
        // .replace('##ORGANIZATION##', 'opentofu-be-ab')
    writeFile file: 'deployer/backend.tf', text: backendRendered

    // 2. Read baseline tfvars from the env folder
    def baseline = readJSON file: "deployer/${dep.env_folder}/terraform.tfvars.json"

    // 3. Read per-deployment config.yml
    def deploymentConfig = readYaml file: "artifacts/${dep.staging}/config.yml"

    // 4. Strip Jenkins-only fields that aren't terraform variables
    def jenkinsOnlyFields = ['workspace', 'tenant_short', 'target_short', 'env_folder']
    def overrides = deploymentConfig.findAll { k, v -> !(k in jenkinsOnlyFields) }

    // 5. Deep-merge: baseline first, config.yml overrides win
    def merged = deepMerge(baseline, overrides)

    // 6. Validate required keys before terraform sees the file.
    //    Adjust this list to whatever your terraform code actually requires.
    def required = ['management_group_name', 'common_tags', 'location']
    for (key in required) {
        if (!merged.containsKey(key) || merged[key] == null) {
            error "Missing required tfvar '${key}' after merging " +
                  "deployer/${dep.env_folder}/terraform.tfvars.json with " +
                  "artifacts/${dep.staging}/config.yml"
        }
    }

    // 7. Write merged tfvars into the deployer working directory.
    //    This is the file terraform will load automatically.
    writeJSON(
        file: 'deployer/terraform.tfvars.json',
        json: merged,
        pretty: 2
    )

    echo "Rendered backend.tf for ${dep.staging}:"
    sh 'cat deployer/backend.tf'
    echo "Final tfvars for ${dep.staging}:"
    sh 'cat deployer/terraform.tfvars.json'
}

// Deep-merge two maps. Nested maps are merged recursively;
// non-map values from override replace base.
def deepMerge(Map base, Map override) {
    def result = [:]
    result.putAll(base)
    override.each { k, v ->
        if (v instanceof Map && base[k] instanceof Map) {
            result[k] = deepMerge(base[k], v)
        } else {
            result[k] = v
        }
    }
    return result
}

// Run terraform via the Makefile, with the right credentials and env vars
// for the given deployment and action ('plan' or 'apply').
def runTerraform(Map dep, String action) {
    withCredentials([
        file(
            credentialsId: "azure-cert-${dep.tenant_short}",
            variable: 'AZURE_CERT'
        ),
        string(
            credentialsId: "azure-cert-password-${dep.tenant_short}",
            variable: 'AZURE_CERT_PASSWORD'
        ),
        string(
            credentialsId: "azure-client-id-${dep.tenant_short}",
            variable: 'AZURE_CLIENT_ID'
        ),
        string(
            credentialsId: "azure-tenant-id-${dep.tenant_short}",
            variable: 'AZURE_TENANT_ID'
        ),
        string(
            credentialsId: "azure-subscription-id-${dep.tenant_short}-${dep.target_short}",
            variable: 'AZURE_SUBSCRIPTION_ID'
        )
    ]) {
        sh """
            set -euo pipefail

            # Backend authentication
            export TF_TOKEN_abc_company_name_net=\$TF_TOKEN_abc_company_name_net

            # Azure provider authentication via service principal cert
            export ARM_CLIENT_ID=\$AZURE_CLIENT_ID
            export ARM_TENANT_ID=\$AZURE_TENANT_ID
            export ARM_SUBSCRIPTION_ID=\$AZURE_SUBSCRIPTION_ID
            export ARM_CLIENT_CERTIFICATE_PATH=\$AZURE_CERT
            export ARM_CLIENT_CERTIFICATE_PASSWORD=\$AZURE_CERT_PASSWORD
            export ARM_USE_CLI=false

            # Run the makefile target. The makefile cd's into deployer/
            # and runs terraform there.
            make ${action}
        """
    }
}

def prepareDeployerWorkdir(Map dep) {
    echo "Preparing deployer workdir for ${dep.staging} using env_folder '${dep.env_folder}'"

    // 1. Hoist backend.tf.tmp -> backend.tf as-is.
    //    The prefix is already baked into the .tmp file; no substitution needed.
    sh "cp deployer/${dep.env_folder}/backend.tf.tmp deployer/backend.tf"

    // 2. Read baseline tfvars from the env folder
    def baseline = readJSON file: "deployer/${dep.env_folder}/terraform.tfvars.json"

    // 3. Read per-deployment config.yml
    def deploymentConfig = readYaml file: "artifacts/${dep.staging}/config.yml"

    // 4. Strip Jenkins-only fields that aren't terraform variables
    def jenkinsOnlyFields = ['workspace_suffix', 'tenant_short', 'target_short', 'env_folder']
    def overrides = deploymentConfig.findAll { k, v -> !(k in jenkinsOnlyFields) }

    // 5. Deep-merge baseline + overrides (config.yml wins)
    def merged = deepMerge(baseline, overrides)

    // 6. Validate required keys
    def required = ['management_group_name', 'common_tags', 'location']
    for (key in required) {
        if (!merged.containsKey(key) || merged[key] == null) {
            error "Missing required tfvar '${key}' after merging " +
                  "deployer/${dep.env_folder}/terraform.tfvars.json with " +
                  "artifacts/${dep.staging}/config.yml"
        }
    }

    // 7. Write merged tfvars into deployer working directory
    writeJSON(
        file: 'deployer/terraform.tfvars.json',
        json: merged,
        pretty: 2
    )

    echo "backend.tf for ${dep.staging}:"
    sh 'cat deployer/backend.tf'
    echo "Final tfvars for ${dep.staging}:"
    sh 'cat deployer/terraform.tfvars.json'
}

def runTerraform(Map dep, String action) {
    withCredentials([
        file(credentialsId: "azure-cert-${dep.tenant_short}", variable: 'AZURE_CERT'),
        string(credentialsId: "azure-cert-password-${dep.tenant_short}", variable: 'AZURE_CERT_PASSWORD'),
        string(credentialsId: "azure-client-id-${dep.tenant_short}", variable: 'AZURE_CLIENT_ID'),
        string(credentialsId: "azure-tenant-id-${dep.tenant_short}", variable: 'AZURE_TENANT_ID'),
        string(credentialsId: "azure-subscription-id-${dep.tenant_short}-${dep.target_short}", variable: 'AZURE_SUBSCRIPTION_ID')
    ]) {
        sh """
            set -euo pipefail

            export TF_TOKEN_abc_company_name_net=\$TF_TOKEN_abc_company_name_net

            export ARM_CLIENT_ID=\$AZURE_CLIENT_ID
            export ARM_TENANT_ID=\$AZURE_TENANT_ID
            export ARM_SUBSCRIPTION_ID=\$AZURE_SUBSCRIPTION_ID
            export ARM_CLIENT_CERTIFICATE_PATH=\$AZURE_CERT
            export ARM_CLIENT_CERTIFICATE_PASSWORD=\$AZURE_CERT_PASSWORD
            export ARM_USE_CLI=false

            # Select the workspace within the prefix the backend.tf declared.
            # -or-create handles first-run for new tenant/target combos.
            export TF_WORKSPACE=${dep.workspace_suffix}

            make ${action}
        """
    }
